package Taller3;


public class Punto1 {

    public static void main(String args [])

        int n=10;
        
        int i, j;
        
        
        for (j = 1; j<=n; j++)
        {
            for (i=1; i<=j; i++)
            {
                
              System.out.print("* ");
            
            }   
            
            System.out.println("");
                
         }      
                
     }
        
    }
